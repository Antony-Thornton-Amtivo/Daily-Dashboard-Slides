///$tab Mapping - MCS Flag Only
MCS_Filter:
load
	Customer_ID,
    Measure,
    ProposalID,
    Customer_ID&ProposalID&Measure as Measure_Key,
    if(left(Measure,3)='MCS','MCS','Other') as MCSFlag,
    if(isnull(#Inspections),0,#Inspections) as #Inspections,
    Year(Calendar_Key) as "Converted year"

FROM [lib://$(vDataFiles)/Transformation2_PAS_Proposals.qvd]
// FROM [lib://$(vDataFilesTest)/Transformation2_PAS_Proposals.qvd]
(qvd)
;


NoConcatenate

MCS_Temp:
load
*
resident MCS_Filter
where MCSFlag = 'MCS';

MCS_Flag:
mapping load 
ProposalID,
MCSFlag
resident MCS_Filter;

MCS_Temp, 
MCS_Filter

///$tab MCS Days
Let TableColumn = 'PASInspectionsDetails';
Let TableName = 'Transformation2_PASInspectionsDetails_MCS';

// CREATE MCS ONLY TABLE - DISTINCT to prevent source duplicates
$(TableName)_MCS:
Load 
    ProposalID,
    // "Proposal PAS BoltOnList",
    SubField("Proposal PAS BoltOnList", ',') AS _SingleEntry,
    ApplyMap('MCS_Flag', ProposalID, 'Other') as MCSFlag
FROM [lib://QVD Files:DataFiles/Transformation1_Proposals.qvd]
(qvd)
;



Left Join ($(TableName)_MCS)
LOAD
    ProposalID,
    count(ProposalID) as TotalInspections
    //Sum(Num(Mid(_SingleEntry, Index(_SingleEntry,'(')+1, Index(_SingleEntry,')')-Index(_SingleEntry,'(')-1))) AS TotalInspections
RESIDENT $(TableName)_MCS
where not isnull(_SingleEntry)
GROUP BY ProposalID;


NoConcatenate

$(TableName):
load 
ProposalID,
MCSFlag,
avg(TotalInspections) as [PAS Inspections]
resident $(TableName)_MCS
where MCSFlag='MCS'
group by ProposalID,MCSFlag
;

store $(TableName) INTO [lib://$(vDataFiles)/$(TableName).qvd] (qvd);
store $(TableName) INTO [lib://$(vDataFilesTest)/$(TableName).qvd] (qvd);


Drop table 
$(TableName)_MCS,
$(TableName)
;



///$tab NonMCS Days
Let TableColumn = 'PASInspectionsDetails';
Let TableName = 'Transformation2_PASInspectionsDetails_Non_MCS';

//CREATE NON MCS DAYS
$(TableName)_Non_MCS:


Load
    RowNo() AS _RowID,
    ProposalID,
    SubField([Proposal PAS BoltOnInspections], ',') AS _SingleEntry,
    ApplyMap('MCS_Flag',ProposalID,'Other') as MCSFlag

FROM [lib://QVD Files:DataFiles/Transformation1_Proposals.qvd]
(qvd)
;


Left Join ($(TableName)_Non_MCS)
LOAD
    _RowID,
    Sum(Num(Mid(_SingleEntry, Index(_SingleEntry,'(')+1, Index(_SingleEntry,')')-Index(_SingleEntry,'(')-1))) AS TotalInspections
RESIDENT $(TableName)_Non_MCS
GROUP BY _RowID;





NoConcatenate

$(TableName):
load 
ProposalID,
MCSFlag,
sum(TotalInspections) as [PAS Inspections]
resident $(TableName)_Non_MCS
where MCSFlag='Other'
group by ProposalID,MCSFlag
;

store $(TableName) INTO [lib://$(vDataFiles)/$(TableName).qvd] (qvd);
// store $(TableName) INTO [lib://$(vDataFilesTest)/$(TableName).qvd] (qvd);




drop table
$(TableName)_Non_MCS,
$(TableName)
;
