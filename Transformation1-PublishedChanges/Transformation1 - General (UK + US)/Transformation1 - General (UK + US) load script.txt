///$tab Auditor Working Days
Let TableName = 'Transformation1_AuditorWorkingDays';
Let TableColumn = 'AuditorWD';

$(TableName)_Temp1:

LOAD
	DataType,
    AuditorID 								as [$(TableColumn) AuditorID],
    //"Auditor Name" 						as [$(TableColumn)],
    Date(AuditorStartDate,'YYYYMMDD') 		as [$(TableColumn) AuditorStartDate],
    Date(AuditorEndDate,'YYYYMMDD') 		as [$(TableColumn) AuditorEndDate],
    "Non Working Days" 						as [$(TableColumn) Non Working Days],
    "Sub Contractor?" 						as [$(TableColumn) Sub Contractor?],
    "Subby Days" 							as [$(TableColumn) Subby Days],
    "Leaving Date" 							as [$(TableColumn) Leaving Date],
    "Formula End Date" 						as [$(TableColumn) Formula End Date],
    "Auditor Category"						as [$(TableColumn) Auditor Category]
    
FROM [lib://$(vDataFiles)/Extract_AuditorWorkingDays.qvd]
// FROM [lib://$(vDataFilesTest)/Extract_AuditorWorkingDays.qvd]
(qvd);



$(TableName):

Load
	*,
	If([$(TableColumn) AuditorIsCompetentFlag] = 1, 4 * [$(TableColumn) Counter Exc Bank Holidays], 0) as [$(TableColumn) PAS Daily Inspections Target]
    ;
    

Load*,
	Applymap('Bank_Holiday_Map',[$(TableColumn) Auditor Date],[$(TableColumn) Counter]) as [$(TableColumn) Counter Exc Bank Holidays],
    ApplyMap('Auditor_Status_Map',[$(TableColumn) AuditorID],'<missing>') as [$(TableColumn) Auditor Status]
    //NetWorkDays([$(TableColumn) AuditorStartDate],[$(TableColumn) AuditorStartDate]) as [$(TableColumn) Working Day Flag]
;


Load*,
	If(SubStringCount([$(TableColumn) Non Working Days], WeekDay_Temp1) >= 1, 0,1) as [$(TableColumn) Counter],
    ApplyMap('AuditorIsPASFlag',[$(TableColumn) AuditorID],'<missing>') as [$(TableColumn) AuditorIsCompetentFlag]
;


Load
	*,
    WeekDay([$(TableColumn) Auditor Date]) as WeekDay_Temp1,
    ;

Load
	DataType,
	[$(TableColumn) AuditorID],
    [$(TableColumn) AuditorStartDate],
    [$(TableColumn) AuditorEndDate],
    [$(TableColumn) Non Working Days],
    [$(TableColumn) Sub Contractor?],
    [$(TableColumn) Subby Days],
    [$(TableColumn) Leaving Date],
    [$(TableColumn) Formula End Date],
    [$(TableColumn) Auditor Category],
    Date(([$(TableColumn) AuditorStartDate] + IterNo() - 1),'YYYYMMDD')      as [$(TableColumn) Auditor Date]
Resident $(TableName)_Temp1


While Date(([$(TableColumn) AuditorStartDate] + IterNo() - 1),'YYYYMMDD') <= [$(TableColumn) AuditorEndDate];

Drop Table $(TableName)_Temp1;


Drop field WeekDay_Temp1;


Store $(TableName) INTO [lib://$(vDataFiles)/$(TableName).qvd] (qvd);
Store $(TableName) INTO [lib://$(vDataFilesTest)/$(TableName).qvd] (qvd);

Let TableName = 'Transformation1_AuditorNACECodes';

$(TableName)_Temp1:
//Load 1 = QMS Data Load
LOAD
    Auditor_ID,
    Au_First,
    Au_Surname,
    Au_SubContract,
    Au_IsFullyActive,
    Au_IsFullyActiveStatus,
    Au_County,

(qvd)
Where Len(Au_Competence_QMS)>0
;

//Load 2 = QMS Data Load

    Auditor_ID,

    Au_Surname,
    Au_SubContract,
    Au_IsFullyActive,
    Au_IsFullyActiveStatus,
    Au_County,
    Au_location,
    Au_Remove,

// FROM [lib://$(vDataFilesTest)/Transformation1_Auditors.qvd]
(qvd)

;

//Load 3 = ISMS Data Load

    Auditor_ID,
    Au_First,
    Au_Surname,
    Au_SubContract,
    Au_IsFullyActive,
    Au_IsFullyActiveStatus,
    Au_County,
    Au_location,

(qvd)

//Load 4 = HSMS Data Load
LOAD
    Auditor_ID,
    Au_First,
    Au_Surname,
    Au_SubContract,
    Au_IsFullyActive,
    Au_IsFullyActiveStatus,
    Au_County,
    Au_location,
    Au_Remove,

    Au_Country,

Where Len(Au_Competence_HSMS)>0
