WithdrawalReason_table:

Load*,
If(VisitDate = 'No visit' and ContractDate = 'No Contract Signed' and NOExpiryDateFlag = 'No Expiry','Yes','No') as NewBizQuoteClient
;


Load*,
If(FinalWithVersion ='2',ApplyMap('map2', WithdrawalReason_ID,'<missing>'),If(FinalWithVersion='0',ApplyMap('map1',(WithdrawalReason_ID),'<missing>'))) as [Withdrawal Reason]
;

Load*,
If(IsNull([Withdrawal ExpiryDate]),'No Expiry') as NOExpiryDateFlag
;

Load*,
If(MATCH([Proc Stat],'WITHDRAWN','WITHDRAWN - New Business','Withdrawal Pending'),'1','0') AS [Current Status]
;

LOAD

	1 as WithdrawalCount,	
    Withdrawal_ID,
    Customer_ID,
    ApplyMap('map3',Customer_ID,'No visit') as VisitDate,
    ApplyMap('map4',Customer_ID,'No Expiry') as [Withdrawal ExpiryDate],
    ApplyMap('map5',Customer_ID,'No Contract Signed') as ContractDate,
    ApplyMap('map6',Customer_ID,'<missing>') as [Company Name],
    ApplyMap('map7',Customer_ID,'<missing>') as [Proc Stat],
    ApplyMap('map8',Customer_ID,'<missing>') as [Sales Advisor],
    ApplyMap('map9',Customer_ID,'<missing>') as [Proposal Type],
    ApplyMap('map10',Customer_ID,'<missing>') as [Proposal Advisor],
    Date(Floor(WithdrawalDate),'DD/MM/YYYY') as [Withdrawal_Calendar_Key (WithdrawalDate)],
    Date(Floor(WithdrawalDate),'DD/MM/YYYY') as [WithdrawalDate],
    Year(WithdrawalDate) as WithdrawalYear,
    Month(WithdrawalDate) as WithdrawalMonth,
    WithdrawalReason_ID,
    WithVersion,
   iF(IsNull(WithVersion),0,WithVersion) as FinalWithVersion
    
FROM [lib://$(vDataFiles)/Extract_tblWithdrawal.qvd]
