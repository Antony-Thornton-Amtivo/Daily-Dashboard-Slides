///$tab Mapping - visits
Load_Visits:
LOAD
    *,
    Customer_ID&Calendar_Key as KeyForVisitDescription
FROM [lib://QVD Files:DataFiles/Transformation2_AuditorVisits.qvd]
(qvd);

Map_LatestVisitDate:
mapping Load
Customer_ID,
Date(Max(Calendar_Key),'DD/MM/YYYY') as LatestVisit
resident Load_Visits
group by Customer_ID
;

Map_VisitDescription:
Mapping Load
KeyForVisitDescription,
VisitDescription
Resident Load_Visits;

///$tab tblComplaintLog
tblComplaintLog_Temp:
LOAD
    ComplaintNature,
    ComplaintManager,
    ComplaintRoot,
    ComplaintCorrective,
    ComplaintStatus,
    ComplaintDateOpen,
    ComplaintDateClosed,
    ComplaintCategory,
    ComplaintAction,
    ComplaintComment,
    Complaint_ID,
    Customer_ID,
    Note_ID
FROM [lib://QVD Files:DataFiles/Extract_tblComplaintLog.qvd]

Customer_ID,

///$tab tblCustomers
tblCustomers_Temp:
load
*,
ApplyMap('Map_VisitDescription',KeyForVisitDescription,'n/a') as VisitDescription,
;

Load
*,
Customer_ID&Date(LatestVisit,'YYYYMMDD') as KeyForVisitDescription
;

LOAD
   *,
   if(ExpiryDate2019>ExpiryDate, ExpiryDate2019,
   	if(ExpiryDate2023>ExpiryDate,ExpiryDate2023,ExpiryDate)) as ExpiryDatePAS,
    SubField([Customer Standard Description], ' + ') AS StandardExploded,
    
    ApplyMap('Map_BabAdvisor',Customer_ID,'Not found') as [SP Bab Advisor],
    Applymap('Map_LatestVisitDate',Customer_ID,'No visit date found for customer') as LatestVisit
    


FROM [lib://QVD Files:DataFiles/Transformation1_Customers.qvd]
(qvd);

left join (tblCustomers_Temp)

tblComplaintLog:
load *
Resident tblComplaintLog_Temp;
;
